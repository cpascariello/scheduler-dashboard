### Task 2: Add formatCpuLabel helper

**Files:**
- Modify: `src/lib/format.ts` (add function at end)

**Step 1: Add formatCpuLabel**

Append to `src/lib/format.ts`:

```typescript
const CPU_VENDOR_LABELS: Record<string, string> = {
  AuthenticAMD: "AMD",
  GenuineIntel: "Intel",
};

export function formatCpuLabel(
  vendor: string | null,
  architecture: string | null,
): string {
  const v = vendor ? (CPU_VENDOR_LABELS[vendor] ?? vendor) : null;
  if (v && architecture) return `${v} ${architecture}`;
  if (v) return v;
  if (architecture) return architecture;
  return "Unknown";
}
```
